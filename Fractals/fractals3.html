This tree randomizes the angles, and draws circles at the end of each branch. 


tree(leonardo, 6, 70, 20)



def tree(t, depth, size, angle):
    if depth == 0:
        t.fd(size)
        t.circle(3)
        t.bk(size)
    else:
        r = random.randrange(1, 3)
        if r == 1:
            angle = angle * 0.75
        elif r == 2:
            angle = angle * 1.2
        elif r == 3:
            angle = angle * 1.50
        t.fd(size)
        t.rt(angle)
        tree(t, depth-1, size * 0.75, angle)
        t.lt(2 * angle)
        tree(t, depth-1, size * 0.75, angle)
        t.rt(angle)
        t.bk(size)
